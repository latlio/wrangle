Using the storms dataset in dplyr – find all the storms that changed status and category during their duration. Here, we’re only interested in their initial and final statuses and categories. Assume that no storm lasted from December of one year to January of the next year, but keep in mind that storm names are re-used across years. Output should have columns: name, initial_date (datetime), final_date (datetime), initial_status (chr), final_status (chr), initial_category (chr), final_category (chr) 
#Enter your code here
library(tidyverse)
library(Lahman)

# Load the Lahman datasets
data("Batting")
data("Salaries")

# Calculate career batting average for each player
career_batting_avg <- Batting %>%
  group_by(playerID) %>%
  summarize(G = sum(G),
            AB = sum(AB),
            H = sum(H)) %>%
  filter(G >= 500) %>%
  mutate(BA = H / AB) %>%
  select(playerID, BA)

# Find players with highest career batting average
highest_ba_player <- career_batting_avg %>%
  filter(BA == max(BA, na.rm = T))

# Get player's hits, at-bats, and home runs
player_stats <- Batting %>%
  filter(playerID %in% highest_ba_player$playerID) %>%
  group_by(playerID) %>% 
  summarize(Hits = sum(H),
            AtBats = sum(AB),
            HR = sum(HR))

# Get player's total salary
player_salary <- Salaries %>%
  filter(playerID %in% highest_ba_player$playerID) %>%
  group_by(playerID) %>% 
  summarize(TotalSalary = sum(salary))

# Join all dataframes
final_df <- highest_ba_player %>%
  left_join(player_stats, by = "playerID") %>%
  left_join(player_salary, by = "playerID") %>%
  select(playerID, BA, Hits, AtBats, HR, TotalSalary)

# Return the final dataframe
final_df
storms %>%
  mutate(date = as.Date(glue('{year}-{month}-{day}'))) %>%
  group_by(year, name) %>% # storm names get re-used across years
  summarise(duration_days = as.numeric(max(date) - min(date))) %>% 
  group_by(year) %>%
  filter(duration_days == max(duration_days)) %>%
  ungroup()
storms %>% 
  mutate(date = as.POSIXct(glue('{year}-{month}-{day} {hour}:00:00'))) %>% 
  group_by(name, year) %>% 
  mutate(initial_final = case_when(date == min(date) ~ "initial",
                                   date == max(date) ~ "final",
                                   T ~ as.character(NA))) %>% 
  ungroup() %>% 
  filter(!is.na(initial_final)) %>% 
  distinct(name, year, date, status, category, initial_final) %>% 
  pivot_wider(id_cols = c(name, year),
              names_from = initial_final,
              values_from = c(date, status, category))
